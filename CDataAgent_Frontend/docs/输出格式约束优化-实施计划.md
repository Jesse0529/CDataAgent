> 状态：全部阶段完成 ✅ | 分支：`dev2.1`  
> 设计基线：`前端输出格式约束优化方案.md` | 后端配套：`CDataAgent_Backend/docs/后端输出格式约束优化-实施计划.md`

## 目标对照

| 目标 | 如何实现 |
|---|---|
| 用户可见回答不依赖 Markdown 语法是否完整 | 渲染 `RenderDocument.blocks`，不解析 Markdown |
| 前端不把模型字符串放入 `v-html` | 区块组件用 Vue 文本节点 `{{ text }}` 渲染 |
| 表格、图表、结论、列表显式区块类型 | `RenderBlock` 判别联合类型，按 type 分发组件 |
| 新旧消息双向兼容 | `if (message.renderDocument)` 分支 + legacy 保留 |
| 断线恢复后一致 | SSE `id` + `(runId, eventId)` 去重 |

### 1.3 新增运行时守卫

**文件**：`/src/utils/renderDocument.ts`（新增）

```ts
// 运行时类型守卫 — 拒绝非法格式、未知版本、字段不完整的文档
export function isValidRenderDocument(doc: unknown): doc is RenderDocument {
  if (!doc || typeof doc !== 'object') return false
  const d = doc as Record<string, unknown>
  if (d.version !== 1) return false
  if (typeof d.runId !== 'string') return false
  if (!Array.isArray(d.blocks)) return false
  return d.blocks.every(isValidBlock)
}

function isValidBlock(block: unknown): block is RenderBlock {
  if (!block || typeof block !== 'object') return false
  const b = block as Record<string, unknown>
  if (typeof b.id !== 'string') return false
  if (typeof b.type !== 'string') return false
  switch (b.type) {
    case 'summary':
    case 'paragraph':
      return typeof b.text === 'string'
    case 'bullets':
      return Array.isArray(b.items) && b.items.every(i => typeof i === 'string')
    case 'table':
      return Array.isArray(b.headers) && Array.isArray(b.rows) && typeof b.totalRows === 'number'
    case 'chart':
      return typeof b.chartIndex === 'number'
    case 'notice':
      return (b.level === 'info' || b.level === 'warning') && typeof b.text === 'string'
    default:
      return false
  }
}

// 安全转换：非法文档返回 null，调用方走 legacy 渲染
export function safeParseRenderDocument(data: unknown): RenderDocument | null {
  if (isValidRenderDocument(data)) return data
  console.warn('[renderDocument] Invalid document rejected:', data)
  return null
}
```

### 4.1 新增渲染分支

**文件**：`/src/components/analysis/ChatMessage.vue`

核心逻辑（伪代码）：

```vue
<template>
  <!-- 现有：用户消息、状态显示等不变 -->

  <!-- AI 消息体：新增分支 -->

      :key="block.id"
      :is="blockComponentMap[block.type]"
      :block="block"
    />

  <!-- 旧协议：保留原 Markdown 渲染 -->
  <div v-else class="ai-body-legacy">
    <!-- 原有的 contentSegments / v-html 逻辑 -->
  </div>

<script setup lang="ts">
import { defineAsyncComponent } from 'vue'
import { isValidRenderDocument } from '@/utils/renderDocument'

const blockComponentMap: Record<string, Component> = {
  summary: defineAsyncComponent(() => import('./blocks/SummaryBlock.vue')),
  paragraph: defineAsyncComponent(() => import('./blocks/ParagraphBlock.vue')),
  bullets: defineAsyncComponent(() => import('./blocks/BulletListBlock.vue')),
  table: defineAsyncComponent(() => import('./blocks/DataTableBlock.vue')),
  chart: defineAsyncComponent(() => import('./blocks/ChartBlock.vue')),
  notice: defineAsyncComponent(() => import('./blocks/NoticeBlock.vue')),
```

关键约束：

- 区块组件全部 `defineAsyncComponent`，不进入首屏 bundle
- `degraded` 为 true 时在顶部显示 NoticeBlock 提示

## 阶段 5：`useAgentStream` 改造

### 5.1 流式状态扩展

**文件**：`/src/composables/useAgentStream.ts`

新增状态：
```ts
const currentRunId = ref<string | null>(null)
const lastEventId = ref<string | null>(null)
const seenEventIds = new Set<string>()    // (runId, eventId) 去重
const pendingDocument = ref<RenderDocument | null>(null)
```

### 5.2 新事件处理

```ts
// onMeta：记录 runId，确认协议版本
function handleMeta(meta: MetaEvent, eventId: string | null) {
  currentRunId.value = meta.runId
  lastEventId.value = eventId
  // 不拼接文本，显示加载态
}

// onDocument：原子设置最终答案
function handleDocument(doc: RenderDocument, eventId: string | null) {
  if (!isDuplicate(currentRunId.value, eventId)) {
    message.renderDocument = doc
    message.renderVersion = 1
    pendingDocument.value = doc
    lastEventId.value = eventId
    markSeen(currentRunId.value, eventId)
  }
}

// onComplete：结束流，不覆盖 document
function handleComplete(event: StructuredEvent, eventId: string | null) {
  message.status = 'done'
  message.tokenUsage = event.tokenUsage
  // 不设置 message.content = event.analysis（v1 下 document 是唯一权威来源）
}
```

### 5.3 旧协议消息处理保留

```ts
// onToken：仅当 renderProtocol !== 'render-document.v1' 时累积
if (renderProtocol.value !== 'render-document.v1') {
  pendingContent += token
  flushBuffer()
}
```

### 5.4 重连改为 `runId + lastEventId`

```ts
// 旧：resumeSeq=${eventCount}
// 新：runId=${currentRunId}&lastEventId=${lastEventId}&resumeToken=${resumeToken}

async function reconnect() {
  const url = `/apis/agent/chat/resume?conversationId=${conversationId}&runId=${currentRunId.value}&lastEventId=${lastEventId.value}&resumeToken=${resumeToken.value}&renderProtocol=render-document.v1`
  // ...
}
```

### 5.5 幂等去重

```ts
function isDuplicate(runId: string | null, eventId: string | null): boolean {
  if (!runId || !eventId) return false
  const key = `${runId}:${eventId}`
  if (seenEventIds.has(key)) return true
  return false
}

function markSeen(runId: string | null, eventId: string | null): void {
  if (runId && eventId) seenEventIds.add(`${runId}:${eventId}`)
}
```
